001package edu.pdx.cs410J.whitlock;
002
003import org.junit.Test;
004import org.xml.sax.SAXException;
005import org.xml.sax.SAXParseException;
006
007import javax.xml.parsers.DocumentBuilder;
008import javax.xml.parsers.DocumentBuilderFactory;
009import javax.xml.parsers.ParserConfigurationException;
010import java.io.IOException;
011
012public class AirlineXmlHelperTest {
013
014  @Test
015  public void canParseValidXmlFile() throws ParserConfigurationException, IOException, SAXException {
016    AirlineXmlHelper helper = new AirlineXmlHelper();
017
018
019    DocumentBuilderFactory factory =
020      DocumentBuilderFactory.newInstance();
021    factory.setValidating(true);
022
023    DocumentBuilder builder =
024      factory.newDocumentBuilder();
025    builder.setErrorHandler(helper);
026    builder.setEntityResolver(helper);
027
028    builder.parse(this.getClass().getResourceAsStream("valid-airline.xml"));
029  }
030
031  @Test(expected = SAXParseException.class)
032  public void cantParseInvalidXmlFile() throws ParserConfigurationException, IOException, SAXException {
033    AirlineXmlHelper helper = new AirlineXmlHelper();
034
035
036    DocumentBuilderFactory factory =
037      DocumentBuilderFactory.newInstance();
038    factory.setValidating(true);
039
040    DocumentBuilder builder =
041      factory.newDocumentBuilder();
042    builder.setErrorHandler(helper);
043    builder.setEntityResolver(helper);
044
045    builder.parse(this.getClass().getResourceAsStream("invalid-airline.xml"));
046  }
047
048}
