001package edu.pdx.cs410J.whitlock;
002
003import edu.pdx.cs410J.lang.Human;
004
005import java.util.HashSet;
006import java.util.Iterator;
007import java.util.LinkedHashSet;
008import java.util.Set;
009
010/**                                                                                 
011 * This class is represents a <code>Student</code>.                                 
012 */                                                                                 
013public class Student extends Human {
014
015  private final double gpa;
016  private final Set<String> classes;
017  private String genderPronoun;
018
019  /**
020   * Creates a new <code>Student</code>                                             
021   *                                                                                
022   * @param name                                                                    
023   *        The student's name                                                      
024   * @param classes                                                                 
025   *        The names of the classes the student is taking.  A student              
026   *        may take zero or more classes.                                          
027   * @param gpa                                                                     
028   *        The student's grade point average                                       
029   * @param gender                                                                  
030   *        The student's gender ("male" or "female", or "other", case insensitive)
031   * @throws IllegalArgumentException
032   *         If the GPA is not between 0.0 and 4.0 or if the gender is unknown
033   */
034  public Student(String name, Set<String> classes, double gpa, String gender) throws IllegalArgumentException {
035    super(name);
036
037    if (gpa > 4.0) {
038      throw new IllegalArgumentException(gpa + " is too big");
039
040    } else if (gpa < 0.0) {
041      throw new IllegalArgumentException(gpa + " is too small");
042    }
043
044    this.classes = classes;
045
046
047    this.gpa = gpa;
048
049    if (gender.equalsIgnoreCase("female")) {
050      this.genderPronoun = "She";
051
052    } else if (gender.equalsIgnoreCase("male")) {
053      this.genderPronoun = "He";
054
055    } else if (gender.equalsIgnoreCase("other")) {
056      this.genderPronoun = "They";
057
058    } else {
059      throw new IllegalArgumentException("Unknown gender: " + gender);
060    }
061  }
062
063  /**
064   * All students say "This class is too much work"
065   */
066  @Override
067  public String says() {                                                            
068    return "This class is too much work";
069  }
070
071  double getGpa() {
072    return this.gpa;
073  }
074
075  /**
076   * Returns a <code>String</code> that describes this
077   * <code>Student</code>.
078   */
079  @Override
080  public String toString() {
081    int numberOfClasses = this.classes.size();
082    StringBuilder sb = new StringBuilder()
083      .append(getName()).append(" has a GPA of ").append(getGpa())
084      .append(" and is taking ").append(numberOfClasses).append(" class");
085    appendNumberOfClassesSuffix(numberOfClasses, sb);
086
087    appendClassNames(sb);
088
089    sb.append("  ");
090    sb.append(getGenderPronoun());
091    sb.append(" says \"");
092    sb.append(says());
093    sb.append("\".");
094
095    return sb.toString();
096  }
097
098  private void appendClassNames(StringBuilder sb) {
099    int numberOfClasses = this.classes.size();
100    if (numberOfClasses == 1) {
101      sb.append(this.classes.iterator().next());
102
103    } else if (numberOfClasses == 2) {
104      Iterator<String> iter = this.classes.iterator();
105      sb.append(iter.next());
106      sb.append(" and ");
107      sb.append(iter.next());
108
109    } else {
110      for (Iterator<String> iter = this.classes.iterator(); iter.hasNext(); ) {
111        String className = iter.next();
112        if (!iter.hasNext()) {
113          sb.append("and ");
114        }
115        sb.append(className);
116        if (iter.hasNext()) {
117          sb.append(", ");
118        }
119      }
120    }
121
122
123    sb.append(".");
124  }
125
126  private void appendNumberOfClassesSuffix(int numberOfClasses, StringBuilder sb) {
127    switch (numberOfClasses) {
128      case 0:
129        sb.append("es.");
130        break;
131      case 1:
132        sb.append(": ");
133        break;
134      default:
135        sb.append("es: ");
136        break;
137    }
138  }
139
140  /**
141   * Main program that parses the command line, creates a
142   * <code>Student</code>, and prints a description of the student to
143   * standard out by invoking its <code>toString</code> method.
144   */
145  public static void main(String[] args) {
146    if (args.length == 0) {
147      printErrorMessageAndExit("Missing command line arguments");
148
149    } else if (args.length == 1) {
150      printErrorMessageAndExit("Missing gender");
151
152    } else if (args.length == 2) {
153      printErrorMessageAndExit("Missing gpa");
154
155    } else {
156      String gpaString = args[2];
157      double gpa;
158      try {
159        gpa = Double.parseDouble(gpaString);
160
161      } catch (NumberFormatException ex) {
162        printErrorMessageAndExit("Invalid GPA: " + gpaString);
163        return;
164      }
165
166      HashSet<String> classes = new LinkedHashSet<>();
167      for (int i = 3; i < args.length; i++) {
168        classes.add(args[i]);
169      }
170
171      Student student;
172      try {
173        student = new Student(args[0], classes, gpa, args[1]);
174
175      } catch (IllegalArgumentException ex) {
176        printErrorMessageAndExit(ex.getMessage());
177        return;
178      }
179
180      System.out.println(student.toString());
181      System.exit(0);
182
183    }
184
185
186  }
187
188  private static void printErrorMessageAndExit(String message) {
189    System.err.println(message);
190    System.exit(1);
191  }
192
193  String getGenderPronoun() {
194    return genderPronoun;
195  }
196}
